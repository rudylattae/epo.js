A thin namespaced localStorage API.

IMPORTANT: tote is not yet stable so you should not use it in production. If you like what you see so far, star it on GitHub so I know you want to use it.

var about = tote('about-tote');

about.set( 'tags', ['simple', 'tiny'] );
console.log( about.get('tags') );
// ['simple', 'tiny']

about.set( 'isActive', true );
console.log( about.get('isActive') );
// true

about.remove('isActive');
console.log( about.get('isActive') );
// null
